/*
  Joint Class Test - Robot Arm
  
  Testing the Joint class with target/speed control
  Commands:
  - S# : Set speed (e.g., S30 for 30°/second)
  - T# : Set target (e.g., T120 for 120°)
*/

#include "Joint.h"

// Create a joint on pin 17
Joint testJoint(17);

void setup() {
  Serial.begin(115200);
  delay(1000);
  Serial.println("Joint Test Starting...");
  Serial.println("Commands: S# (set speed), T# (set target)");
  Serial.println("Examples: S30, T120, S0, T45");
  Serial.println("Joint initialized on pin 17");
}

void loop() {
  // CRITICAL: Update the joint continuously
  testJoint.update();
  delay(10);  // Small delay to avoid flooding the serial output
  
  // Check for serial commands
  if (Serial.available() > 0) {
    String input = Serial.readStringUntil('\n');
    input.trim();
    input.toUpperCase();
    
    if (input.length() >= 2) {
      char command = input.charAt(0);
      float value = input.substring(1).toFloat();
      
      if (command == 'S') {
        testJoint.speed = value;
        Serial.print("Speed set to: ");
        Serial.print(value);
        Serial.println("°/second");
        Serial.print("Current angle: ");
        Serial.print(testJoint.currentAngle);
        Serial.print("°, Target: ");
        Serial.print(testJoint.target);
        Serial.println("°");
      }
      else if (command == 'T') {
        testJoint.target = value;
        Serial.print("Target set to: ");
        Serial.print(value);
        Serial.println("°");
        Serial.print("Current angle: ");
        Serial.print(testJoint.currentAngle);
        Serial.print("°, Speed: ");
        Serial.print(testJoint.speed);
        Serial.println("°/s");
      }
      else {
        Serial.println("Error: Unknown command");
        Serial.println("Use S# for speed, T# for target");
      }
    }
    else {
      Serial.println("Error: Invalid command format");
      Serial.println("Examples: S30, T120");
    }
  }
}
